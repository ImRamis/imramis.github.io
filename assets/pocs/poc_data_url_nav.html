// Test 7: MIME type confusion
function test7() {
  var frame = document.getElementById('t7frame');
  frame.style.display = 'block';

  const mimeTests = [
    { mime: 'text/html', content: '' },
    { mime: 'application/xhtml+xml', content: '' },
    { mime: 'image/svg+xml', content: '' },
    { mime: 'application/javascript', content: 'parent.postMessage({type:"t7",mime:"application/javascript",exec:true},"*")' },
  ];

  log('t7', 'Testing script execution across MIME types...');

  let i = 0;
  function nextMimeTest() {
    if (i >= mimeTests.length) {
      log('t7', 'All MIME tests complete.');
      return;
    }
    const test = mimeTests[i];
    const url = 'data:' + test.mime + ',' + encodeURIComponent(test.content);
    frame.src = url;
    log('t7', `Testing ${test.mime}...`);
    i++;
    setTimeout(nextMimeTest, 1000);
  }

  window.addEventListener('message', (e) => {
    if (e.data && e.data.type === 't7') {
      log('t7', `  ✓ ${e.data.mime}: SCRIPT EXECUTED`);
    }
  });

  // Set timeout to detect non-execution
  mimeTests.forEach((test, idx) => {
    setTimeout(() => {
      // This will log for all, but execution messages will appear before
      log('t7', `  [${test.mime}] timeout check (no execution message = blocked)`);
    }, (idx + 1) * 1000 + 500);
  });

  nextMimeTest();
}
